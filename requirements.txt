#    pip install -r requirements.txt

# 3. Create directory structure:
#    Secret_couples/
#    ├── head_detector.py
#    ├── staring.py
#    ├── utils/
#    │   ├── __init__.py
#    │   ├── pose_estimation.py
#    │   └── tracker.py
#    ├── VIDEOS/
#    │   └── input_video.mp4  # Your video file here
#    ├── STARE_FRAMES/  # Will be created automatically
#    └── logs/  # Will be created automatically

# 4. Update video path in head_detector.py:
#    VIDEO_PATH = r"C:\Users\DELL\OneDrive\Documents\Secret_couples\VIDEOS\your_video.mp4"

# 5. Run the system:
#    python head_detector.py

# 6. Controls during execution:
#    - 'q': Quit
#    - 'p': Pause/Resume
#    - 'd': Show debug information

# Solution: Check VIDEO_PATH in head_detector.py
